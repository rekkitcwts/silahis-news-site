-- Get all articles, name of the author, the title and the body:
SELECT staff.staff_lname, staff.staff_fname, articletype.type_name, status.status_name, article.article_title, article.date_approved, article.article_body FROM article INNER JOIN staff ON (article.author = staff.staff_id) INNER JOIN articletype ON (article.article_type = articletype.type_id) INNER JOIN status ON (article.status = status.status_id) ORDER BY article.article_id DESC, article.date_approved DESC
